### p的指向性问题
``` js
function serObj(p){
    p.name="小红";
    p={};
    p.name="西西"
}
var p={
    name:"小明"
}
setObj(p); //小红
console.log(p);
```
**解析**
``` js
function serObj(p){
    p.name="小红"; //通过p找到小明，
    //--将原来的小明改名为小红
    p={};//创建新对象开辟了一个新的内存
    //--至此局部变量p和外面存"小明"的
    //--那块内存没关系了
    p.name="西西"
}
var p={
    name:"小明"
}
setObj(p); //小红
console.log(p);
```
